logrotate 分割nginx 日志

  • /opt/f1/logs/*.log {
  •      daily
  •      dateext
  •      missingok
  •      rotate 90
  •      notifempty
  •      compress
  •      sharedscripts
  •      postrotate
  •          [ -f /opt/f1/logs/nginx.pid ] && kill -USR1 `cat /opt/f1/logs/nginx.pid`
  •      endscript
  • }
