keepalived 配置

安装

  • yum -y install keepalived
  • mv /etc/keepalived/keepalived.conf /etc/keepalived/keepalived.conf.ori

主节点配置文件

  • cat >> /etc/keepalived/keepalived.conf << EOF
  • ! Configuration File for keepalived
  • global_defs {
  •    notification_email {
  •      liqilin@huanqiu.com
  •    }
  •    notification_email_from tech-ops@huanqiu.com
  •    smtp_server 192.168.205.2
  •    smtp_connect_timeout 30
  •    router_id  Uclou_odn
  • }
  • vrrp_script check {     #定义脚本
  •    script "curl --connect-timeout 2 -x 127.0.0.1:80 w.huanqiu.com" 
  •    interval 10 
  •    weight 100  
  • }
  • vrrp_instance VI_1 {
  •     state MASTER
  •     interface eth0
  •     virtual_router_id 11
  •     priority 150
  •     advert_int 1
  •     authentication {
  •         auth_type PASS
  •         auth_pass huanqiu18
  •     }
  •     virtual_ipaddress {
  •      192.168.205.26 
  •     }
  • }
  • vrrp_instance VI_2 {
  •     state BACKUP
  •     interface eth0
  •     virtual_router_id 12
  •     priority 100
  •     advert_int 1
  •     authentication {
  •         auth_type PASS
  •         auth_pass huanqiu181
  •     }
  •     virtual_ipaddress {
  •      192.168.205.25 
  •     }
  • }
  • EOF

从节点配置文件

  • cat >> /etc/keepalived/keepalived.conf << EOF
  • ! Configuration File for keepalived
  • global_defs {
  •    notification_email {
  •      liqilin@huanqiu.com
  •    }
  •    notification_email_from tech-ops@huanqiu.com

  •    smtp_connect_timeout 30
  •    router_id  Uclou_odn
  • }
  • vrrp_script checkcurl {
  •    script "/usr/bin/curl --connect-timeout 2 -x 127.0.0.1:80 w.huanqiu.com"
  •    interval 10
  •    weight -60
  • }
  • vrrp_instance VI_1 {
  •     state MASTER
  •     interface eth0
  •     virtual_router_id 11

  •     advert_int 1
  •     authentication {
  •         auth_type PASS
  •         auth_pass huanqiu18
  •     }
  •     virtual_ipaddress {
  •      192.168.205.26 
  •     }
  •     track_script
  •     {
  •         checkcurl
  •     }
  • }
  • vrrp_instance VI_2 {
  •     state BACKUP
  •     interface eth0
  •     virtual_router_id 12
  •     priority 150
  •     advert_int 1
  •     authentication {
  •         auth_type PASS

  •     }
  •     virtual_ipaddress {
  •      192.168.205.25 
  •     }
  •     track_script
  •     {
  •         checkcurl
  •     }
  • }
  • EOF
